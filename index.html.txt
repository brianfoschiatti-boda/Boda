    <!-- Script de JavaScript para la función de copiar -->
    <script>
        function copiarAlias() {
            const alias = document.getElementById('alias-texto').innerText;
            
            navigator.clipboard.writeText(alias).then(() => {
                const boton = document.getElementById('btn-copiar');
                boton.innerText = '¡Copiado! ✓';
                boton.style.backgroundColor = '#6E6359'; 
                
                setTimeout(() => {
                    boton.innerText = 'Copiar Alias';
                    boton.style.backgroundColor = '#8C7B6E';
                }, 2000);
            }).catch(err => {
                alert('No se pudo copiar automáticamente. Por favor, seleccionalo manualmente.');
            });
        }
    </script>
